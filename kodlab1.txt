package ru.mirea.hash;

import org.jfree.chart.ChartFactory;
import org.jfree.chart.JFreeChart;
import org.jfree.data.category.DefaultCategoryDataset;

import javax.imageio.ImageIO;
import java.awt.*;
import java.awt.image.BufferedImage;
import java.io.File;
import java.io.IOException;
import java.util.*;

public class BarSample {

    public static void main(String[] args) throws IOException {
        HashMap<String, Integer> map = new HashMap<>();
        try (Scanner s=new Scanner(new File("1.txt"),"Cp1251")) {
            while (s.hasNext()){
                String word=s.next();
                if (!map.containsKey(word)) {
                    map.put(word, 1);
                }
                else {
                    int count = map.get(word);
                    map.put(word,++count);
                }

            }
        }

        Set<Map.Entry<String, Integer>> set = map.entrySet();
        ArrayList<Map.Entry<String, Integer>> list = new ArrayList<>(set);
        list.sort((e1,e2)->-e1.getValue().compareTo(e2.getValue()));
        for (Map.Entry<String, Integer> entry : list) {
            System.out.println(entry);
        }

        DefaultCategoryDataset data = new DefaultCategoryDataset();
        String category = "Население";
        int count=0;
        for (Map.Entry<String, Integer> entry : list) {
            count++;
            if (count<=10){
            data.addValue(entry.getValue(), category, entry.getKey());}
            else break;
        }
//        data.addValue(142, category, "и");
//        data.addValue(94, category, "в");
//        data.addValue(68, category, "не");
        JFreeChart chart = ChartFactory.createBarChart("Встречаемость слов", "Слово", "Встречаемость, шт", data);
        BufferedImage image = chart.createBufferedImage(600, 400);
        File file = new File("chart.png");
        ImageIO.write(image, "png", file);
        Desktop.getDesktop().open(file);
    }
}
